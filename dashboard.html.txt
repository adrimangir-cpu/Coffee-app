<div class="card" id="content">Загрузка...</div>

<script>
    const FB_URL = "https://project-lenta22-default-rtdb.firebaseio.com";
    let currentCatalog = {};

    async function loadTab(type, btn) {
        document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
        const box = document.getElementById('content');
        box.innerHTML = "⌛ Синхронизация...";

        if (type === 'catalog') {
            let res = await fetch(`${FB_URL}/catalogs/order.json`);
            currentCatalog = await res.json() || {};
            let html = "<h3>Редактор справочника</h3><table><tr><th>Категория</th><th>Товар</th><th>Цена</th><th>Действие</th></tr>";
            for (let k in currentCatalog) {
                if (currentCatalog[k].name.toLowerCase().includes("товар")) continue;
                html += `<tr><td>${currentCatalog[k].cat}</td><td><b>${currentCatalog[k].name}</b></td>
                <td><input type="number" value="${currentCatalog[k].price}" onchange="currentCatalog['${k}'].price=this.value"></td>
                <td><button class="btn-del" onclick="deleteItem('${k}')">Удалить</button></td></tr>`;
            }
            box.innerHTML = html + "</table><button class='btn-save' onclick='saveCatalog()'>💾 СОХРАНИТЬ ИЗМЕНЕНИЯ</button>";
        } else {
            let res = await fetch(`${FB_URL}/orders.json`);
            let orders = await res.json() || {};
            let html = "<table><tr><th>Точка</th><th>Дата</th><th>Детали</th></tr>";
            for (let k in orders) {
                let items = Object.entries(orders[k].items).map(([n,q]) => `${n}(${q})`).join(', ');
                html += `<tr><td><b>${orders[k].shop}</b></td><td>${orders[k].date}</td><td>${items}</td></tr>`;
            }
            box.innerHTML = html + "</table>";
        }
    }

    function deleteItem(id) { delete currentCatalog[id]; loadTab('catalog', document.querySelector('.tab-btn.active')); }
    async function saveCatalog() {
        await fetch(`${FB_URL}/catalogs/order.json`, { method: 'PUT', body: JSON.stringify(currentCatalog) });
        alert("Справочник обновлен!");
    }
    
    window.onload = () => loadTab('orders', document.querySelector('.tab-btn'));
</script>
